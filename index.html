1Подключение к MySQL + таблица users
📁 connect.h — настройка соединения с базой данных (порт 3307, пароль usbw). Создаётся таблица users с полями: id, login, password(MD5), role, try_count, is_blocked. Автоматически добавляется admin (пароль 123). Функция db::connect() возвращает true при успехе.

bool db::connect()
{
    QSqlDatabase db = QSqlDatabase::addDatabase("QMYSQL");
    QTextCodec::setCodecForLocale(QTextCodec::codecForName("UTF-8"));
    db.setHostName("localhost");
    db.setPort(3307);
    db.setUserName("root");
    db.setPassword("usbw");
    db.setDatabaseName("prod");

    if (!db.open()) {
        qDebug() << "Database error:" << db.lastError().text();
        return false;
    }

    QSqlQuery query;
    query.exec("CREATE TABLE IF NOT EXISTS users ("
               "id INT PRIMARY KEY AUTO_INCREMENT, "
               "login VARCHAR(50) UNIQUE NOT NULL, "
               "password VARCHAR(32) NOT NULL, "
               "role VARCHAR(50) DEFAULT 'Пользователь', "
               "try_count INT DEFAULT 0, "
               "is_blocked TINYINT DEFAULT 0)");

    query.exec("INSERT IGNORE INTO users (login, password) VALUES ('admin', '202cb962ac59075b964b07152d234b70')");

2Кастомная таблица — MyTableWidget (drag-drop)
💡 Реализует внутреннее перетаскивание фрагментов пазла. Переопределены mousePressEvent, mouseMoveEvent, dropEvent. При перетаскивании создаётся визуальная миниатюра, фрагменты меняются местами. Теперь drag-and-drop работает идеально.
📄 mytablewidget.h
#ifndef MYTABLEWIDGET_H
#define MYTABLEWIDGET_H

#include <QtWidgets>

class MyTableWidget : public QTableWidget
{
    Q_OBJECT
public:
    explicit MyTableWidget(QWidget *parent = nullptr);

protected:
    void dragEnterEvent(QDragEnterEvent *event) override;
    void dragMoveEvent(QDragMoveEvent *event) override;
    void dropEvent(QDropEvent *event) override;
    void mousePressEvent(QMouseEvent *event) override;
    void mouseMoveEvent(QMouseEvent *event) override;

private:
    QPoint dragStartPosition;
    QTableWidgetItem *draggedItem = nullptr;
    int draggedRow = -1;
    int draggedCol = -1;
};

#endif
⚙️ mytablewidget.cpp
#include "mytablewidget.h"

MyTableWidget::MyTableWidget(QWidget *parent)
    : QTableWidget(parent)
{
    setAcceptDrops(true);
    setDragEnabled(true);
    setDropIndicatorShown(true);
    setSelectionMode(QAbstractItemView::SingleSelection);
    setDragDropMode(QAbstractItemView::InternalMove);
}

void MyTableWidget::mousePressEvent(QMouseEvent *event)
{
    if (event->button() == Qt::LeftButton) {
        dragStartPosition = event->pos();
        QModelIndex index = indexAt(event->pos());
        if (index.isValid()) {
            draggedRow = index.row();
            draggedCol = index.column();
            draggedItem = item(draggedRow, draggedCol);
        }
    }
    QTableWidget::mousePressEvent(event);
}

void MyTableWidget::mouseMoveEvent(QMouseEvent *event)
{
    if (!(event->buttons() & Qt::LeftButton)) return;
    if ((event->pos() - dragStartPosition).manhattanLength() < QApplication::startDragDistance()) return;
    if (!draggedItem) return;

    QDrag *drag = new QDrag(this);
    QMimeData *mimeData = new QMimeData;
    QByteArray data;
    QDataStream stream(&data, QIODevice::WriteOnly);
    stream << draggedRow << draggedCol << draggedItem->text();
    mimeData->setData("application/x-puzzle-piece", data);
    drag->setMimeData(mimeData);

    QPixmap pixmap(90,90);
    pixmap.fill(Qt::transparent);
    QPainter painter(&pixmap);
    QIcon icon = draggedItem->icon();
    if (!icon.isNull()) painter.drawPixmap(5,5,icon.pixmap(80,80));
    else painter.fillRect(5,5,80,80,QBrush(QColor(200,200,200)));
    drag->setPixmap(pixmap);
    drag->setHotSpot(QPoint(45,45));
    drag->exec(Qt::MoveAction);
    QTableWidget::mouseMoveEvent(event);
}

void MyTableWidget::dragEnterEvent(QDragEnterEvent *event) {
    if(event->mimeData()->hasFormat("application/x-puzzle-piece")) event->acceptProposedAction();
}
void MyTableWidget::dragMoveEvent(QDragMoveEvent *event) {
    if(event->mimeData()->hasFormat("application/x-puzzle-piece")) event->acceptProposedAction();
}
void MyTableWidget::dropEvent(QDropEvent *event)
{
    if(!event->mimeData()->hasFormat("application/x-puzzle-piece")) return;
    QPoint dropPos = event->pos();
    QModelIndex targetIndex = indexAt(dropPos);
    if(!targetIndex.isValid()) return;
    int targetRow = targetIndex.row(), targetCol = targetIndex.column();
    QByteArray data = event->mimeData()->data("application/x-puzzle-piece");
    QDataStream stream(&data, QIODevice::ReadOnly);
    int sourceRow, sourceCol; QString text;
    stream >> sourceRow >> sourceCol >> text;
    if(sourceRow==targetRow && sourceCol==targetCol) return;
    QTableWidgetItem *sourceItem = takeItem(sourceRow, sourceCol);
    QTableWidgetItem *targetItem = takeItem(targetRow, targetCol);
    if(sourceItem && targetItem) {
        setItem(sourceRow, sourceCol, targetItem);
        setItem(targetRow, targetCol, sourceItem);
    } else if(sourceItem) {
        setItem(targetRow, targetCol, sourceItem);
        setItem(sourceRow, sourceCol, nullptr);
    }
    event->acceptProposedAction();
    draggedItem = nullptr; draggedRow = -1; draggedCol = -1;
}
3Auth: логин + пазл-капча
🧩 Класс Auth проверяет пазл (правильный порядок 1,2,3,4), хеширует пароль MD5, считает неудачные попытки и блокирует пользователя после 3 ошибок. Важно: виджет MyTableWidget интегрирован в .ui форму. Метод setupPuzzleTable инициализирует drag-drop и случайное перемешивание.

void Auth::setupPuzzleTable()
{
    ui->tableWidget->setRowCount(2); ui->tableWidget->setColumnCount(2);
    ui->tableWidget->horizontalHeader()->setVisible(false);
    ui->tableWidget->verticalHeader()->setVisible(false);
    ui->tableWidget->setIconSize(QSize(100,100));
    ui->tableWidget->setDragEnabled(true);
    ui->tableWidget->setAcceptDrops(true);
    ui->tableWidget->setDropIndicatorShown(true);
    ui->tableWidget->setDragDropMode(QAbstractItemView::InternalMove);
    for(int col=0;col<2;++col) ui->tableWidget->setColumnWidth(col,100);
    for(int row=0;row<2;++row) ui->tableWidget->setRowHeight(row,100);
    shufflePuzzle();
}

void Auth::shufflePuzzle()
{
    std::vector<int> numbers = {1,2,3,4};
    unsigned seed = std::chrono::system_clock::now().time_since_epoch().count();
    std::shuffle(numbers.begin(), numbers.end(), std::default_random_engine(seed));
    int idx=0;
    for(int r=0;r<2;r++) for(int c=0;c<2;c++) {
        QTableWidgetItem *item = new QTableWidgetItem();
        int num = numbers[idx];
        QString path = QString(":/puzzle/%1.png").arg(num);
        QIcon icon(path);
        if(icon.isNull()) { QPixmap pix(80,80); pix.fill(QColor(100+num*30,100+num*20,200)); item->setIcon(QIcon(pix)); }
        else item->setIcon(icon);
        item->setText(QString::number(num));
        item->setFlags(Qt::ItemIsEnabled | Qt::ItemIsSelectable | Qt::ItemIsDragEnabled | Qt::ItemIsDropEnabled);
        ui->tableWidget->setItem(r,c,item);
        idx++;
    }
}
bool Auth::isPuzzleCorrect() {
    std::vector<int> orig = {1,2,3,4}, cur;
    for(int r=0;r<2;r++) for(int c=0;c<2;c++) cur.push_back(ui->tableWidget->item(r,c)->text().toInt());
    return cur == orig;
}
void Auth::resetPuzzle() { shufflePuzzle(); }
bool Auth::isUserBlocked(const QString& login) {
    if(blockedUsers.value(login,false)) return true;
    QSqlQuery q; q.prepare("SELECT is_blocked FROM users WHERE login=:login"); q.bindValue(":login",login);
    if(q.exec() && q.next()) { bool b = q.value(0).toBool(); blockedUsers[login]=b; return b; }
    return false;
}
void Auth::blockUser(const QString& login) { 
    blockedUsers[login]=true; 
    QSqlQuery q; q.prepare("UPDATE users SET is_blocked=1 WHERE login=:login"); q.bindValue(":login",login); q.exec();
}
void Auth::incrementFailedAttempts(const QString& login) {
    failedAttempts[login] = failedAttempts.value(login,0)+1;
    if(failedAttempts[login]>=3) blockUser(login);
}
void Auth::resetFailedAttempts(const QString& login) { failedAttempts[login]=0; }

void Auth::on_enterButton_clicked() {
    QString login = ui->loginEdit->text(), pass = ui->passEdit->text();
    if(login.isEmpty() || pass.isEmpty()) { QMessageBox::warning(this,"Ошибка","Заполните поля"); return; }
    if(isUserBlocked(login)) { QMessageBox::critical(this,"Доступ запрещен","Вы заблокированы"); return; }
    if(!isPuzzleCorrect()) {
        QMessageBox::warning(this,"Капча","Соберите пазл правильно!");
        incrementFailedAttempts(login);
        if(failedAttempts[login]>=3) QMessageBox::critical(this,"Блокировка","Вы заблокированы");
        else resetPuzzle();
        return;
    }
    QString pass_hash = QCryptographicHash::hash(pass.toUtf8(), QCryptographicHash::Md5).toHex();
    QSqlQuery query;
    query.prepare("SELECT id FROM users WHERE login=:login AND password=:password");
    query.bindValue(":login",login); query.bindValue(":password",pass_hash);
    if(query.exec() && query.next()) {
        resetFailedAttempts(login);
        MainWindow *mw = new MainWindow(query.value(0).toInt(), login);
        mw->show(); this->close();
    } else {
        incrementFailedAttempts(login);
        QMessageBox::critical(this,"Ошибка","Неверный логин/пароль. Осталось попыток: "+QString::number(3-failedAttempts[login]));
    }
}
